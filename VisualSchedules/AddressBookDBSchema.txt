CREATE TABLE Person (    pid INTEGER PRIMARY KEY,    first_name STRING,    mid_name STRING    last_name STRING,    sex STRING,    email STRING,    phone STRING,    fax STRING);CREATE TABLE Company (    cid INTEGER PRIMARY KEY,    cname STRING,    contact_person STRING,    email STRING,    phone STRING,    fax STRING,);	CREATE TABLE Address (    aid INTEGER PRIMARY KEY,    street_no STRING,    street_name STRING,    city STRING,    state STRING,    zip STRING,    country STRING,    );CREATE TABLE Person_Address (    pid INTEGER,    aid INTEGER,    tag ENUM ('home', 'company', 'other'),    PRIMARY KEY(pid, aid),    FOREIGN KEY pid REFERENCES Person(pid),    FOREIGN KEY aid REFERENCES Address(aid),);CREATE TABLE Company_Address (    cid INTEGER,    aid INTEGER,    PRIMARY KEY(cid, aid),    FOREIGN KEY cid REFERENCES Company(cid),    FOREIGN KEY aid REFERENCES Address(aid));
